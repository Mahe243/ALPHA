#include<iostream>
using namespace std;
int binarysearch(int arr[],int N,int x){
int low=1,high=N;
while(low<=high){
int mid = (low+high)/2;
if(x==arr[mid]){
return mid;}
else if(x<arr[mid]){
high = mid-1;}
else{
low=mid+1;}
return -1;
}
}

Best case: o(1)
Average and Worst case: o(logN)
